#!/usr/bin/env python3 """ collatz_graph.py Generate a 2D projection of the 3D Collatz (x, y, z) structure: - x = starting integer n - y = T^k(n) (Collatz value at step k) - z = k (step index) We plot (X, Y) with: - X = n - Y = T^k(n) + 0.1 * k This shows: - positive funnel to 1 via 16 -> 8 -> 4 -> 2 -> 1 - negative loops like -5 <-> -7 and -17 cycle """ import matplotlib.pyplot as plt def collatz(n): """Yield the Collatz trajectory starting from n.""" seen = set() while True: yield n if n in seen: # loop detected return seen.add(n) if n == 1 or n == -1: return if n % 2 == 0: n //= 2 else: n = 3 * n + 1 def main(): X_pos, Y_pos = [], [] X_neg, Y_neg = [], [] # Range of starting values for start in range(-50, 51): k = 0 for val in collatz(start): X = start Y = val + 0.1 * k # 2D projection of (x, y, z) if start >= 0: X_pos.append(X) Y_pos.append(Y) else: X_neg.append(X) Y_neg.append(Y) k += 1 plt.figure(figsize=(10, 6)) # Positive trajectories (blue) plt.scatter(X_pos, Y_pos, s=5, c="blue", label="Positive starts") # Negative trajectories (red) plt.scatter(X_neg, Y_neg, s=5, c="red", label="Negative starts") # Highlight special points plt.scatter([-5, -7], [-7, -5], s=40, c="red") plt.text(-5, -7, "-5 ↔ -7", color="red", fontsize=9) plt.scatter([-17], [-17], s=40, c="red") plt.text(-17, -17, "-17 loop", color="red", fontsize=9) plt.scatter([17], [1], s=40, c="blue") plt.text(17, 1, "17 → 1", color="blue", fontsize=9) for p in [16, 8, 4, 2, 1]: plt.scatter([p], [p], s=30, c="blue") plt.text(p, p, str(p), color="blue", fontsize=8) plt.title("2D Projection of 3D Collatz Graph") plt.xlabel("x = starting value n") plt.ylabel("y = T^k(n) + 0.1 k") plt.grid(True) plt.legend() plt.tight_layout() plt.savefig("collatz_graph.png", dpi=300) plt.show() if __name__ == "__main__": main()
